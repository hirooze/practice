####################
# Git Command List #
####################

# コミットファイルの有無確認
git status

# ファイルをステージングする
git add <file>
## example
## git add -A
## git add <dir>/*

# ローカルリポジトリにコミットする
git commit -m 'message'

# ローカルリポジトリをリモートリポジトリへpushする
git push origin master


##################
# Others Command #
##################

# リモートリポジトリの詳細を確認する
git remote -v

# リモートリポジトリを登録する
git remote add <name> <url>

# リモートリポジトリを削除する
git remote rm <name>

# ブランチの一覧を表示する
git branch

# ブランチを作成する
git branch <branchname>
## example
## git branch devs
## [作成+切り替え]　git checkout -b <branch> 

# ブランチを切り替える
git checkout <branchname>
## example
## git checkout devs

# ブランチを削除する
git branch -d <branchname> 
## example
## git branch -d devs
